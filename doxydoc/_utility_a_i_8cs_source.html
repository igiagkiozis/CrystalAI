128  bool InternalAddBehaviour(Behaviour behaviour) {
129  if(string.IsNullOrEmpty(behaviour.NameId) ||
130  _behaviourMap.ContainsKey(behaviour.NameId))
131  return false;

133  _behaviourMap.Add(behaviour.NameId, behaviour);

139  bool InternalAddBehaviour(string behaviourId) {
140  if(_behaviourMap.ContainsKey(behaviourId))
141  return false;
142 
143  var behaviour = _collection.Behaviours.Create(behaviourId) as Behaviour;
144  _behaviourMap.Add(behaviourId, behaviour);
145  _behaviours.Add(behaviour);
